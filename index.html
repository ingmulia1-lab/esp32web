<script>


  const broker = "wss://3759e23402c4448584ca5389cdd951e9.s1.eu.hivemq.cloud:8884/mqtt";


  const options = {

    username: "Ivan Mulia",

    password: "Musiol1980",

    reconnectPeriod: 1000,

    clean: true

  };


  const topicControl = "ivan/esp32/led";

  const topicEstado = "ivan/esp32/estado";


  const client = mqtt.connect(broker, options);


  client.on("connect", function () {

      document.getElementById("conexion").innerText = "Conectado ✅";

      client.subscribe(topicEstado);

  });


  client.on("error", function (err) {

      document.getElementById("conexion").innerText = "Error ❌";

      console.log(err);

  });


  client.on("message", function (topic, message) {

      if(topic === topicEstado){

          const estado = message.toString();

          if(estado === "ON"){

              document.getElementById("led").style.background = "lime";

          }else{

              document.getElementById("led").style.background = "red";

          }

      }

  });


  function encender(){

      client.publish(topicControl, "ON");

  }


  function apagar(){

      client.publish(topicControl, "OFF");

  }


</script>
